Java

This code snippet is a simple program to make an input title-cased without using regular expressions.

package com.company;

import java.util.Scanner;

public class Main {

    public static void main(String[] args) {

        String input;
        System.out.println("Please enter a string");
        Scanner scan = new Scanner(System.in);
        input = scan.nextLine();
        System.out.println("The string has been title-cased to: " + caps(input));

    }

    public static String caps(String _input) {

        String output;
        String newString = "";
        output = _input.toLowerCase();
        String[] array = output.split(" ");
        for (int i = 0; i < array.length; i++){
            newString = newString.concat(array[i].toUpperCase().charAt(0) + array[i].substring(1) + " ");
        }
        return newString;
    }
}

This code snippet does some fancy animations and styling.

$(function() {
	$('li').hide().slideDown(1000);
	$('#animals').html($('.enabled').length);
	$('#add-btn').on('click', function() {

		$('#add-btn').hide();
		$('input').fadeIn(1000);
		$('button').fadeIn(1000);
		$('#validate').fadeIn(1000);
	});
	
	$('button').click(function(){

		if ($('input').val().length < 3){
			$('#validate').html("ERORR: Enter more than 3 characters!");
		} else {
			$('ul').append('li class="enabled"' + $('input').val() + '/li');
			$('#animals').html($('.enabled').length);
			$('input').val("");
			$('#validate').html('');
		}
	});
	
	$('ul').on('click', 'li', function() {

		if ($(this).hasClass('complete')) {
			$(this).effect('blind').delay(500).queue(function() { $(this).remove(); });
		} else {
			$(this).appendTo('ul');
			$(this).removeClass();
			$(this).addClass('complete');
			$('#animals').html($('.enabled').length);
		}
	});
});

This code snippet is a small section cut out of a whole program that emulates a battle between monsters and humans.

using System;
using System.Collections.Generic;
using System.IO;

namespace FinalPractical
{
    class CombatEnvironment
    {
        //probably need to store the combatants
        public List combatantsList = new List();
        public string fileOutput = "";
        public string finalOutput = "";

        public CombatEnvironment(List combatantsParam)
        {
            //sorting combatants by speed would be a good idea
            for (int i = 0; i < combatantsParam.Count - 1; i++)
            {
                for (int j = i + 1; j > 0; j--)
                {
                    if (combatantsParam[j - 1].Speed < combatantsParam[j].Speed)
                    {
                        Character temp = combatantsParam[j - 1];
                        combatantsParam[j - 1] = combatantsParam[j];
                        combatantsParam[j] = temp;
                    }
                }
            }

            foreach (Character c in combatantsParam)
            {
                combatantsList.Add(c);
            }

        }

        public void ResolveCombat()
        {
            //to start, display information about all combatants
            //when displaying information include the type of Character(Monster or Player), health, attack, defense, and speed
            int playerCount = 0;
            int monsterCount = 0;

            foreach (Character c in combatantsList)
            {
                if (c is Monster)
                {
                    playerCount++;
                }
                else
                {
                    monsterCount++;
                }

                Console.WriteLine(c.ToString());
                fileOutput += c.ToString() + "\n";
            }

            Console.WriteLine("");
            fileOutput += " \n";

            //every round each Character should attack one other Character of the opposing type (Players attack Monsters, and Monsters attack Players)
            //The Characters should attack in the order of highest speed to lowest speed
            //If a Character's health falls to 0 or below 0 then it has died and should be removed from the CombatEnvironment
            //After each round of combat each Character's information should be displayed with updated health values
            //After each round of combat, check to see if all Players or all Monsters have died. If one side has died, end the combat and display which side was victorious

            List deadChars = new List();

            while (playerCount > 0 && monsterCount > 0)
            {
                foreach (Character c in combatantsList)
                {
                    if (c is Monster && c.Health > 0)
                    {
                        for (int i = 0; i < combatantsList.Count; i++)
                        {
                            if (combatantsList[i] is Player && combatantsList[i].Health > 0)
                            {
                                c.Combat(combatantsList[i]);
                                fileOutput += Character.getCombatOutput();
                                if (combatantsList[i].Health <= 0)
                                {
                                    deadChars.Add(combatantsList[i]);
                                }
                                break;
                            }
                        }
                    }
                    else if (c is Player && c.Health > 0)
                    {
                        for (int i = 0; i < combatantsList.Count; i++)
                        {
                            if (combatantsList[i] is Monster && combatantsList[i].Health > 0)
                            {
                                c.Combat(combatantsList[i]);
                                fileOutput += Character.getCombatOutput();
                                if (combatantsList[i].Health <= 0)
                                {
                                    deadChars.Add(combatantsList[i]);
                                }
                                break;
                            }
                        }
                    }
                }

                for (int i = 0; i < combatantsList.Count; i++)
                {
                    foreach (Character c in deadChars)
                    {
                        if (combatantsList[i].Name.Equals(c.Name))
                        {
                            if (combatantsList[i] is Player)
                            {
                                playerCount--;
                            }
                            else
                            {
                                monsterCount--;
                            }
                            combatantsList.Remove(combatantsList[i]);
                        }
                    }
                }

                Console.WriteLine("Combat Round Results:");
                fileOutput += "Combat Round Results:\n";
                foreach (Character c in combatantsList)
                {
                    Console.WriteLine(c.ToString());
                    fileOutput += c.ToString() + "\n";
                }

                Console.WriteLine("");
                fileOutput += " \n";

                using (StreamWriter txt = File.AppendText("log.txt"))
                {
                    txt.Write(fileOutput);
                }
                fileOutput = "";
            }

            if (playerCount > monsterCount)
            {
                Console.WriteLine("Players Win!");
                finalOutput += "Players Win!";
            } 
            else
            {
                Console.WriteLine("Monsters Win!");
                finalOutput += "Monsters Win!";
            }

            using (StreamWriter txt = File.AppendText("log.txt"))
            {
                txt.Write(finalOutput);
            }
        }
    }
}
